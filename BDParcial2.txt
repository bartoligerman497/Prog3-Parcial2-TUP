create database CasaMusica
go

go

create table tipos
(
id int,
nombre varchar(30)
constraint pk_tipo primary key (id)
)
create table instrumentos
(
id int identity(1,1),
nombre varchar(50),
descripcion varchar(50),
stock int,
precio float(53),
idTipo int
constraint pk_instrumentos primary key (id)
constraint fk_instrumentos_tipos foreign key (idTipo) references tipos (id)
)
